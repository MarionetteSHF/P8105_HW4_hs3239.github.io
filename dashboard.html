```{r setup, include=FALSE}

library(tidyverse)
library(p8105.datasets)
library(plotly)
library(flexdashboard)

```

```{r data_selection}
data("rest_inspec")

rest_inspec =
  rest_inspec %>%
  mutate(boro = str_to_title(boro)) %>% 
  filter(grade %in% c("A", "B", "C"), boro == "Manhattan") %>%
  relocate(boro) %>% 
  sample_n(10000) %>% 
  view
```

rest_inspec %>%
  mutate(text_label = str_c("Score: ", score, "\nGrade: ", grade)) %>% 
  plot_ly(
